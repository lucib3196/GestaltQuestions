class WallHeatTransfer: def __init__(self, thickness_brick, k_brick, thickness_insulation, k_insulation, thickness_concrete, k_concrete, T1, T2): self.thickness_brick = thickness_brick self.k_brick = k_brick self.thickness_insulation = thickness_insulation self.k_insulation = k_insulation self.thickness_concrete = thickness_concrete self.k_concrete = k_concrete self.T1 = T1 self.T2 = T2 self.validate_parameters() def validate_parameters(self): # Validate thickness and thermal conductivity parameters if any(param <= 0 for param in [self.thickness_brick, self.k_brick, self.thickness_insulation, self.k_insulation, self.thickness_concrete, self.k_concrete]): raise ValueError("All thicknesses and conductivities must be positive values.") def calculate_resistance(self, thickness, k): return thickness / k def calculate_total_resistance(self): R_brick = self.calculate_resistance(self.thickness_brick, self.k_brick) R_insulation = self.calculate_resistance(self.thickness_insulation, self.k_insulation) R_concrete = self.calculate_resistance(self.thickness_concrete, self.k_concrete) return R_brick + R_insulation + R_concrete def calculate_heat_transfer(self): R_total = self.calculate_total_resistance() if R_total <= 0: raise ValueError("Total thermal resistance must be positive.") Q = (1 * (self.T1 - self.T2)) / R_total return Q # Example usage try: wall = WallHeatTransfer(thickness_brick=0.2, k_brick=0.72, thickness_insulation=0.1, k_insulation=0.04, thickness_concrete=0.15, k_concrete=1.7, T1=20, T2=-10) heat_transfer = wall.calculate_heat_transfer() print(f'The rate of heat transfer through the wall is approximately {heat_transfer:.2f} W.') except ValueError as error: print(f'Error: {error}')
